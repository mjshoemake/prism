Chart.js Configuration

Here's how to configure Chart.js with your purple and black theme: 

// Chart.js theme configuration
const chartConfig = {
  type: 'bar',
  data: {
    labels: ['Jan', 'Feb', 'Mar', 'Apr', 'May'],
    datasets: [{
      label: 'Sales',
      data: [12, 19, 3, 5, 2],
      backgroundColor: '#6f42c1', // Primary purple
      borderColor: '#5a2d9e',     // Primary dark
      borderWidth: 1
    }]
  },
  options: {
    responsive: true,
    plugins: {
      legend: {
        labels: {
          color: '#ffffff' // White text
        }
      }
    },
    scales: {
      y: {
        beginAtZero: true,
        ticks: { color: '#b8a9d9' }, // Light purple
        grid: { color: '#4a2c7a' }   // Secondary purple
      },
      x: {
        ticks: { color: '#b8a9d9' },
        grid: { color: '#4a2c7a' }
      }
    }
  }
};
